Question 4 : Comment les titres des grands joueurs saccumulent en fonction de leur ge selon leur genre ?

q4 = left_join(grand_slams, detailsJoueur, by = c("joueur")) %>% 
  mutate(age = as.numeric(dateTournoi - dateNaissance)/365) %>%
  group_by(joueur, age, genre) %>% 
  summarise(counts = n()) %>% 
  group_by(joueur) %>% 
  mutate(victoireCumul = cumsum(counts)) %>% 
  arrange(desc(victoireCumul))
## `summarise()` has grouped output by 'joueur', 'age'. You can override using the `.groups` argument.
q4 %>% ggplot(aes(x = age, y = victoireCumul, group = joueur,color=genre)) + geom_point() + geom_step()+geom_rug() + facet_wrap(~genre)+labs(title = "Evolution du palmars de chaque joueur en fonction de son ge")
